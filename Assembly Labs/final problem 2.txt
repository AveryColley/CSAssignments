.ORIG x3000

START
LEA R0, ENTER_PROMPT
PUTS
GETC
OUT
LD R1, NUMBER_CHECK
NOT R1, R1
ADD R1, R1, #1
ADD R2, R0, R1
BRn LOWER
BRz EQUAL
BRp HIGHER

LOWER
LD R0, NEWLINE
OUT
LEA R0, LOWER_RESPONSE
PUTS
LD R0, NEWLINE
OUT
OUT
BRnzp START

EQUAL
LD R0, NEWLINE
OUT
LEA R0, EQUAL_RESPONSE
PUTS
LD R0, NEWLINE
OUT
OUT
BRnzp START

HIGHER
LD R0, NEWLINE
OUT
LEA R0, HIGHER_RESPONSE
PUTS
LD R0, NEWLINE
OUT
OUT
BRnzp START

NEWLINE .FILL x000A
NUMBER_CHECK .FILL x0035
ENTER_PROMPT .STRINGZ "Please enter a number: "
LOWER_RESPONSE .STRINGZ "The number you entered was less than 5!"
EQUAL_RESPONSE .STRINGZ "The number you entered was equal to 5!"
HIGHER_RESPONSE .STRINGZ "The number you entered was greater than 5!"
.END
