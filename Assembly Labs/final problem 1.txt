.ORIG x3000

START

LD R2, CAPITAL_STRING
LD R5, CAPITAL_CORRECTION

NOT R5, R5
ADD R5, R5, #1

GETS

GETC
OUT
ADD R1, R0, #-10
BRz DONE
ADD R0, R5, R0
STR R0, R2, #0
ADD R2, R2, #1
BRnzp GETS 

DONE

AND R3, R3, #0
STR R3, R2, #0
LEA R0, PROMPT
PUTS
LD R0, CAPITAL_STRING
PUTS
LD R0, NEWLINE
OUT
BRnzp START

PROMPT .STRINGZ "You entered: "
NEWLINE .FILL x000A
CAPITAL_CORRECTION .FILL x0020
CAPITAL_STRING .FILL x3100
.END
